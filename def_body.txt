"""
style_cable.py
1. Crea/actualiza un material visual (OmniPBR) tipo goma/poliuretano oscuro
   para que el Cable se vea como una manguera de borescopio.
2. Recalibra el material físico deformable a rangos realistas de goma/
   poliuretano (silicona-like), manteniendo tu densidad que ya estaba bien.
"""

import omni.usd
from pxr import UsdShade, Sdf, Gf, PhysxSchema
from omni.physx.scripts import deformableUtils, physicsUtils

stage = omni.usd.get_context().get_stage()

CABLE_PATH = "/World/Cable"
VISUAL_MATERIAL_PATH = "/World/Looks/BorescopeCableLook"
PHYSICS_MATERIAL_PATH = "/World/Looks/DeformableMaterial_v2"  # el que ya tenías

# ============ 1. Material visual: goma/poliuretano oscuro ============
material = UsdShade.Material.Define(stage, VISUAL_MATERIAL_PATH)
shader = UsdShade.Shader.Define(stage, VISUAL_MATERIAL_PATH + "/Shader")

# IMPORTANTE: OmniPBR es un modulo MDL, no un shader id tipo UsdPreviewSurface.
# Hay que apuntarlo al archivo .mdl real, si no, Omniverse muestra el
# material de "error" (rojo/magenta) porque nunca resuelve el shader.
shader.SetSourceAsset(Sdf.AssetPath("OmniPBR.mdl"), "mdl")
shader.SetSourceAssetSubIdentifier("OmniPBR", "mdl")
shader.GetImplementationSourceAttr().Set(UsdShade.Tokens.sourceAsset)

# Color casi negro con un toque de gris azulado (look de goma/poliuretano)
shader.CreateInput("diffuse_color_constant", Sdf.ValueTypeNames.Color3f).Set(
    Gf.Vec3f(0.035, 0.035, 0.04)
)
# No metálico (es goma, no metal desnudo)
shader.CreateInput("metallic_constant", Sdf.ValueTypeNames.Float).Set(0.0)
# Semi-mate: la goma real tiene un poco de brillo pero no es espejo
shader.CreateInput("reflection_roughness_constant", Sdf.ValueTypeNames.Float).Set(0.55)
# Un poco de specular para que no se vea completamente plano bajo luz
shader.CreateInput("specular_level", Sdf.ValueTypeNames.Float).Set(0.5)

material.CreateSurfaceOutput().ConnectToSource(shader.ConnectableAPI(), "surface")

cable_prim = stage.GetPrimAtPath(CABLE_PATH)
UsdShade.MaterialBindingAPI(cable_prim).Bind(material)
print(f"Material visual '{VISUAL_MATERIAL_PATH}' aplicado a {CABLE_PATH}")

# ============ 2. Recalibrar el material físico deformable ============
physics_mat_prim = stage.GetPrimAtPath(PHYSICS_MATERIAL_PATH)
if not physics_mat_prim.IsValid():
    raise RuntimeError(f"No encontré el material físico en {PHYSICS_MATERIAL_PATH}")

deformable_mat_api = PhysxSchema.PhysxDeformableBodyMaterialAPI(physics_mat_prim)

# Densidad: ya estaba bien (goma/poliuretano real ronda 1100-1300 kg/m3)
deformable_mat_api.GetDensityAttr().Set(1300.0)

# Young's Modulus: lo bajamos un poco para que se sienta más flexible,
# como una manguera real en vez de algo casi rígido. Ajusta entre
# 1e7 (más flexible) y 8e7 (más firme) según qué tanto quieras que se doble.
deformable_mat_api.GetYoungsModulusAttr().Set(30000000.0)  # 30 MPa

# Poisson's Ratio: la goma real es casi incompresible (0.47-0.49).
# Ojo: valores muy cerca de 0.5 pueden hacer la simulación más inestable
# numéricamente, así que 0.47 es un buen punto medio seguro.
deformable_mat_api.GetPoissonsRatioAttr().Set(0.47)

# Fricción dinámica: goma sobre metal/superficies típicas
deformable_mat_api.GetDynamicFrictionAttr().Set(0.45)

# Damping: dejamos el tuyo, ya estaba razonable para que no oscile demasiado
deformable_mat_api.GetElasticityDampingAttr().Set(0.005)
deformable_mat_api.GetDampingScaleAttr().Set(0.5)

print(f"Material físico '{PHYSICS_MATERIAL_PATH}' recalibrado a valores de goma/poliuretano.")
print("Corre la simulación y compara cómo se siente la flexibilidad ahora.")

"""
fix_attachment.py
Crea (o repara) el attachment entre el Cable deformable y el Cube ancla,
asegurando que:
  1. El Cube tenga collider + sea rígido (kinematic, para poder moverlo
     a mano si quieres, o estático si no lo vas a mover nunca).
  2. El attachment tenga Actor0/Actor1 apuntando a los dos prims correctos.
  3. El AutoAttachmentAPI esté configurado para generar los puntos de unión.
"""

import omni.usd
from pxr import PhysxSchema, UsdPhysics

stage = omni.usd.get_context().get_stage()

CABLE_PATH = "/World/Cable"
ANCHOR_PATH = "/World/Cube"          # el objeto fijo al que se pega el cable
ANCHOR_IS_KINEMATIC = True           # True = lo puedes mover a mano/script; False = 100% estático

cable_prim = stage.GetPrimAtPath(CABLE_PATH)
anchor_prim = stage.GetPrimAtPath(ANCHOR_PATH)

if not cable_prim.IsValid():
    raise RuntimeError(f"No encontré {CABLE_PATH}")
if not anchor_prim.IsValid():
    raise RuntimeError(f"No encontré {ANCHOR_PATH}")

# 1. Asegurar que el Cube tenga collider
if not anchor_prim.HasAPI(UsdPhysics.CollisionAPI):
    UsdPhysics.CollisionAPI.Apply(anchor_prim)
    print(f"CollisionAPI aplicado a {ANCHOR_PATH}")

# Asegurar que sea un rigid body (kinematic si lo quieres poder mover)
if not anchor_prim.HasAPI(UsdPhysics.RigidBodyAPI):
    rigid_api = UsdPhysics.RigidBodyAPI.Apply(anchor_prim)
else:
    rigid_api = UsdPhysics.RigidBodyAPI(anchor_prim)

rigid_api.CreateKinematicEnabledAttr().Set(ANCHOR_IS_KINEMATIC)
print(f"RigidBodyAPI en {ANCHOR_PATH} (kinematic={ANCHOR_IS_KINEMATIC})")

# 2. Borrar attachment viejo si existía mal configurado, y crear uno limpio
attachment_path = f"{CABLE_PATH}/attachment"
if stage.GetPrimAtPath(attachment_path).IsValid():
    stage.RemovePrim(attachment_path)
    print("Attachment anterior eliminado.")

attachment = PhysxSchema.PhysxPhysicsAttachment.Define(stage, attachment_path)
attachment.GetActor0Rel().SetTargets([CABLE_PATH])
attachment.GetActor1Rel().SetTargets([ANCHOR_PATH])

# 3. Configurar el auto-attachment para que calcule los puntos de contacto
auto_attach = PhysxSchema.PhysxAutoAttachmentAPI.Apply(attachment.GetPrim())
auto_attach.CreateEnableDeformableVertexAttachmentsAttr().Set(True)
auto_attach.CreateEnableRigidSurfaceAttachmentsAttr().Set(True)
auto_attach.CreateDeformableVertexOverlapOffsetAttr().Set(0.15)
auto_attach.CreateRigidSurfaceSamplingDistanceAttr().Set(0.15)

print(f"Attachment creado entre {CABLE_PATH} y {ANCHOR_PATH}.")
print("Corre la simulación (Play) y revisa si ahora sí queda fijo.")







"""
create_deformable_cable.py

Script para Isaac Sim (Window > Script Editor, o standalone).
Crea un cilindro con alta subdivisión longitudinal (para que el render mesh
pueda seguir la curva del sim mesh) y le aplica el Deformable Body con
los parámetros que ya venías usando.

Ajusta las variables en la sección CONFIG antes de correrlo.
"""

import omni.kit.commands
import omni.usd
from pxr import UsdGeom, Gf, PhysxSchema, UsdPhysics
from omni.physx.scripts import deformableUtils, physicsUtils

# ============ CONFIG - ajusta esto a tu caso ============
PRIM_PATH = "/World/Cable"                 # ruta del nuevo prim
LENGTH_SEGMENTS = 50                        # subdivisión a lo largo del eje (sube esto si sigue viéndose rígido)
RADIAL_SEGMENTS = 10                        # subdivisión radial (no necesita ser muy alta)
TRANSLATE = Gf.Vec3f(7.50849, 7.63377, 8.57485)   # copia la que ya tenías si quieres el mismo lugar
ORIENT_XYZ = Gf.Vec3f(94.568, 48.748, -14.845)     # tus valores de Orient actuales
SCALE = Gf.Vec3f(0.3, 0.3, 5.0)             # tu escala actual (radio, radio, largo)

SIMULATION_HEX_RESOLUTION = 25              # ya tenías 25, está bien para empezar
SOLVER_POSITION_ITERATIONS = 16
SELF_COLLISION = False

# Parámetros de material deformable (los que ya tenías configurados)
DENSITY = 1300.0
DYNAMIC_FRICTION = 0.25
YOUNGS_MODULUS = 50000000.0
POISSONS_RATIO = 0.42
DAMPING_SCALE = 0.5
ELASTICITY_DAMPING = 0.005
# =========================================================

stage = omni.usd.get_context().get_stage()

# 1. Crear el cilindro con alta subdivisión
#    Nota: los nombres de parámetro pueden variar ligeramente según la versión
#    de la extensión omni.kit.primitive.mesh instalada. Si algún kwarg da error,
#    revisa Create > Mesh > Settings en la UI para confirmar el nombre exacto.
success, cylinder_path = omni.kit.commands.execute(
    "CreateMeshPrimWithDefaultXform",
    prim_type="Cylinder",
    prim_path=PRIM_PATH,
    u_patches=RADIAL_SEGMENTS,
    v_patches=LENGTH_SEGMENTS,
    half_scale=None,
)

if not success:
    raise RuntimeError("No se pudo crear el cilindro - revisa los parámetros en la UI (Create > Mesh > Settings)")

cylinder_prim = stage.GetPrimAtPath(cylinder_path)

# 2. Aplicar la transform (mismo lugar/orientación/escala que ya tenías)
#    NOTA: CreateMeshPrimWithDefaultXform ya crea sus propios xformOps.
#    Usamos XformCommonAPI en vez de Add*Op() para reusar esos ops
#    existentes en lugar de chocar con ellos (eso causaba el crash en
#    AddScaleOp y por eso el cilindro se quedaba con el tamaño default).
xform_api = UsdGeom.XformCommonAPI(cylinder_prim)
xform_api.SetTranslate(Gf.Vec3d(TRANSLATE[0], TRANSLATE[1], TRANSLATE[2]))
xform_api.SetRotate(
    (ORIENT_XYZ[0], ORIENT_XYZ[1], ORIENT_XYZ[2]),
    UsdGeom.XformCommonAPI.RotationOrderXYZ,
)
xform_api.SetScale(SCALE)

# 3. Crear el material deformable con tus valores actuales
material_path = "/World/Looks/DeformableMaterial_v2"
deformableUtils.add_deformable_body_material(
    stage,
    material_path,
    density=DENSITY,
    dynamic_friction=DYNAMIC_FRICTION,
    youngs_modulus=YOUNGS_MODULUS,
    poissons_ratio=POISSONS_RATIO,
    damping_scale=DAMPING_SCALE,
    elasticity_damping=ELASTICITY_DAMPING,
)

# 4. Aplicar el Deformable Body al cilindro nuevo
deformableUtils.add_physx_deformable_body(
    stage,
    cylinder_path,
    simulation_hexahedral_resolution=SIMULATION_HEX_RESOLUTION,
    collision_simplification=True,
    self_collision=SELF_COLLISION,
    solver_position_iteration_count=SOLVER_POSITION_ITERATIONS,
)

# 5. Asignar el material físico deformable al prim
physicsUtils.add_physics_material_to_prim(
    stage,
    cylinder_prim,
    material_path,
)

print(f"Cable creado en {cylinder_path} con {LENGTH_SEGMENTS} segmentos a lo largo del eje.")
print("Ahora vuelve a crear el attachment (fijación al extremo) igual que antes,")
print("y corre la simulación para comparar contra el cilindro anterior.")

import omni.usd
from pxr import UsdGeom, PhysxSchema, Vt, Gf

def fix_cable_to_cube_exact(cable_path="/World/Cylinder", cube_path="/World/Cube"):
    stage = omni.usd.get_context().get_stage()
    
    cyl_prim = stage.GetPrimAtPath(cable_path)
    cube_prim = stage.GetPrimAtPath(cube_path)
    mesh = UsdGeom.Mesh(cyl_prim)
    points = mesh.GetPointsAttr().Get()
    
    cube_world_pos = UsdGeom.Xformable(cube_prim).ComputeLocalToWorldTransform(0).ExtractTranslation()
    print(f"Posición real del Cube en el mundo: {cube_world_pos}")
    
    cyl_world_to_local = UsdGeom.Xformable(cyl_prim).ComputeLocalToWorldTransform(0).GetInverse()
    cube_pos_in_cyl_space = cyl_world_to_local.Transform(cube_world_pos)
    # Convertir a Vec3f para que coincida con el tipo de points
    cube_pos_in_cyl_space = Gf.Vec3f(cube_pos_in_cyl_space[0], cube_pos_in_cyl_space[1], cube_pos_in_cyl_space[2])
    print(f"Posición del Cube en espacio local del cilindro: {cube_pos_in_cyl_space}")
    
    distances = [((p - cube_pos_in_cyl_space).GetLength(), i) for i, p in enumerate(points)]
    distances.sort(key=lambda x: x[0])
    
    n_closest = 15
    closest_indices = [i for _, i in distances[:n_closest]]
    closest_points = [points[i] for i in closest_indices]
    print(f"Vértices más cercanos al Cube: {n_closest}, distancia mínima: {distances[0][0]:.4f}")
    
    cyl_local_to_world = UsdGeom.Xformable(cyl_prim).ComputeLocalToWorldTransform(0)
    world_points = [cyl_local_to_world.Transform(p) for p in closest_points]

def fix_cable_to_cube_exact(cable_path="/World/Cylinder", cube_path="/World/Cube"):
    stage = omni.usd.get_context().get_stage()
    
    cyl_prim = stage.GetPrimAtPath(cable_path)
    cube_prim = stage.GetPrimAtPath(cube_path)
    mesh = UsdGeom.Mesh(cyl_prim)
    points = mesh.GetPointsAttr().Get()
    
    # 1. Posición REAL del cubo en el mundo (esto es lo que pides)
    cube_world_pos = UsdGeom.Xformable(cube_prim).ComputeLocalToWorldTransform(0).ExtractTranslation()
    print(f"Posición real del Cube en el mundo: {cube_world_pos}")
    
    # 2. Convertir esa posición mundial al espacio LOCAL del cilindro
    cyl_world_to_local = UsdGeom.Xformable(cyl_prim).ComputeLocalToWorldTransform(0).GetInverse()
    cube_pos_in_cyl_space = cyl_world_to_local.Transform(cube_world_pos)
    print(f"Posición del Cube en espacio local del cilindro: {cube_pos_in_cyl_space}")
    
    # 3. Encontrar los vértices del cilindro MÁS CERCANOS a esa posición exacta
    distances = [((p - cube_pos_in_cyl_space).GetLength(), i) for i, p in enumerate(points)]
    distances.sort(key=lambda x: x[0])
    
    # Tomar los N más cercanos (ajusta este número si necesitas más/menos puntos)
    n_closest = 15
    closest_indices = [i for _, i in distances[:n_closest]]
    closest_points = [points[i] for i in closest_indices]
    print(f"Vértices más cercanos al Cube: {n_closest}, distancia mínima: {distances[0][0]:.4f}")
    
    # 4. Convertir esos puntos a world space, luego a espacio local del cubo
    cyl_local_to_world = UsdGeom.Xformable(cyl_prim).ComputeLocalToWorldTransform(0)
    world_points = [cyl_local_to_world.Transform(p) for p in closest_points]
    
    cube_world_to_local = UsdGeom.Xformable(cube_prim).ComputeLocalToWorldTransform(0).GetInverse()
    points_in_cube_space = [cube_world_to_local.Transform(p) for p in world_points]
    
    # 5. Aplicar al attachment
    attachment_prim = stage.GetPrimAtPath(f"{cable_path}/attachment")
    auto_attach = PhysxSchema.PhysxAutoAttachmentAPI(attachment_prim)
    auto_attach.CreateEnableRigidSurfaceAttachmentsAttr().Set(True)
    auto_attach.CreateEnableDeformableVertexAttachmentsAttr().Set(True)
    auto_attach.CreateDeformableVertexOverlapOffsetAttr().Set(0.15)
    auto_attach.CreateRigidSurfaceSamplingDistanceAttr().Set(0.15)
    
    print(f"✅ Fijado usando posición REAL del Cube — {n_closest} vértices más cercanos")

import omni.usd
from pxr import UsdGeom, PhysxSchema, Gf, Vt

stage = omni.usd.get_context().get_stage()

# Borrar el fixed joint que no sirvió
joint_prim = stage.GetPrimAtPath("/World/fixed_joint")
if joint_prim.IsValid():
    stage.RemovePrim("/World/fixed_joint")
    print("🗑️ Fixed joint eliminado")

# Recalcular attachment correctamente
cyl_prim = stage.GetPrimAtPath("/World/Cylinder")
mesh = UsdGeom.Mesh(cyl_prim)
points = mesh.GetPointsAttr().Get()

z_max = max(p[2] for p in points)
tolerance = 0.05
top_local_points = [p for p in points if p[2] >= z_max - tolerance]
print(f"Puntos superiores (local space): {len(top_local_points)}")

# Convertir a world space usando el transform del cilindro
cyl_xform = UsdGeom.Xformable(cyl_prim)
world_transform = cyl_xform.ComputeLocalToWorldTransform(0)
world_points = [world_transform.Transform(p) for p in top_local_points]

attachment_prim = stage.GetPrimAtPath("/World/Cylinder/attachment")
attachment_api = PhysxSchema.PhysxPhysicsAttachment(attachment_prim)

# Points0 = puntos del cilindro (local space, como ya estaba)
attachment_api.GetPoints0Attr().Set(Vt.Vec3fArray(top_local_points))

# Points1 = LOS MISMOS puntos en world space (posición absoluta para pegar al cubo)
# Convertir a local space del Cube
cube_prim = stage.GetPrimAtPath("/World/Cube")
cube_xform = UsdGeom.Xformable(cube_prim)
cube_world_to_local = cube_xform.ComputeLocalToWorldTransform(0).GetInverse()
points_in_cube_space = [cube_world_to_local.Transform(p) for p in world_points]

attachment_api.GetPoints1Attr().Set(Vt.Vec3fArray(points_in_cube_space))

print(f"✅ Points1 seteado con {len(points_in_cube_space)} puntos coincidentes")

import omni.usd
from pxr import UsdGeom, PhysxSchema, Gf, Vt

stage = omni.usd.get_context().get_stage()

cyl_path = "/World/Cylinder"
cyl_prim = stage.GetPrimAtPath(cyl_path)
mesh = UsdGeom.Mesh(cyl_prim)

points = mesh.GetPointsAttr().Get()
print(f"Total de puntos en el mesh: {len(points)}")

# Encontrar el Z máximo (extremo superior) en local space
z_values = [p[2] for p in points]
z_max = max(z_values)
print(f"Z máximo (extremo superior): {z_max}")

# Seleccionar solo los puntos cerca de ese extremo (tolerancia de 0.05)
tolerance = 0.05
top_indices = [i for i, p in enumerate(points) if p[2] >= z_max - tolerance]
print(f"Puntos en el extremo superior: {len(top_indices)} de {len(points)}")

# Obtener el attachment existente
attachment_prim = stage.GetPrimAtPath(f"{cyl_path}/attachment")
attachment_api = PhysxSchema.PhysxPhysicsAttachment(attachment_prim)

# Definir SOLO esos puntos como los que se pegan (Points Actor 0)
top_points = Vt.Vec3fArray([points[i] for i in top_indices])
attachment_api.GetPoints0Attr().Set(top_points)

print(f"✅ Attachment limitado a {len(top_indices)} puntos del extremo superior")
